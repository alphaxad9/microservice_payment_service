docker build -t alphaxad9/payment-service:0.0.5 .

# 2. Also tag as latest
docker tag alphaxad9/payment-service:0.0.5 alphaxad9/payment-service:latest

# 3. Push to Docker Hub
docker push alphaxad9/payment-service:0.0.5
docker push alphaxad9/payment-service:latest

# 4. Load into Kind
kind load docker-image alphaxad9/payment-service:0.0.5 --name payment-service

# 5. Verify it's loaded
docker exec -it payment-service-control-plane crictl images | grep payment-service



# Apply the updated config
kubectl apply -k services/payment-service/overlays/dev

# Restart the deployment
kubectl rollout restart deployment payment-service -n payment-service

# Watch the pod come up
kubectl get pods -n payment-service
